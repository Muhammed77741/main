    # 3. –ù–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ (bullish/bearish)

    upper_wick = candle['high'] - max(candle['open'], candle['close'])

    score = 0

    # –í—ã—Å–æ–∫–∏–π –æ–±—ä—ë–º = +30

    elif volume_strength > 1.2:

    if body_ratio > 0.7:

    if total_wicks < candle_range * 0.3:
        score += 30

    return score, volume_strength, body_ratio, is_bullish
